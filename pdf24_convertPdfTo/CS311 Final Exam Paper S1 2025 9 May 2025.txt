Section C: Case Study        20 Marks 

Consider a simple bank account with an initial balance. Multiple threads will simulate deposit 
and withdrawal transactions on this shared account concurrently. 

#include <stdio.h> 
#include <stdlib.h> 
#include <pthread.h> 
#include <unistd.h> 
#include <sys/wait.h> 
 
double account_balance = 1000.0; 
pthread_mutex_t balance_mutex; 
 
void *transaction(void *arg) {    char *type = (char *)arg; 
    pthread_mutex_lock(&balance_mutex); 
    if (strcmp(type, "deposit") == 0) { 
        account_balance += 100.0; 
        printf("Deposited. Bal: %.2f (P:%d,T:%lu)\n", account_balance, getpid(), 
pthread_self()); 
    } else if (strcmp(type, "withdraw") == 0) { 
        if (account_balance >= 200.0) { 
            account_balance -= 200.0; 
            printf("Withdrew. Bal: %.2f (P:%d,T:%lu)\n", account_balance, getpid(), 
pthread_self()); 
        } else { 
            printf("Insufficient funds (P:%d,T:%lu)\n", getpid(), pthread_self()); 
        } 
    } 
    pthread_mutex_unlock(&balance_mutex); 
    return NULL; 
} 
 
int main() { 
    pthread_t threads[4]; // Increased to 4 threads 
    char *types[4] = {"deposit", "withdraw", "deposit", "withdraw"}; 
    int i; 
    pid_t child_pid; 
 
    pthread_mutex_init(&balance_mutex, NULL); 
    child_pid = fork(); 
 
    if (child_pid == 0) { 
        // Child: Only withdrawals 
        pthread_t child_threads[2]; 
        char *child_types[2] = {"withdraw", "withdraw"}; 
        for (i = 0; i < 2; i++) { 
            pthread_create(&child_threads[i], NULL, transaction, (void *)child_types[i]); 
        } 
        for (i = 0; i < 2; i++) { 
            pthread_join(child_threads[i], NULL); 
        } 
        pthread_mutex_destroy(&balance_mutex); 
        exit(0); 
}

        }
